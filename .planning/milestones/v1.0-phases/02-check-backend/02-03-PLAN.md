**`in_range` — DO NOT copy Polars verbatim:** Polars uses `compare_min.and_(compare_max)`; narwhals uses `is_between` with `closed`:
```python
_CLOSED_MAP = {
    (True, True): "both",
    (True, False): "left",
    (False, True): "right",
    (False, False): "none",
}

def in_range(data, min_value, max_value, include_min=True, include_max=True):
    closed = _CLOSED_MAP[(include_min, include_max)]
    return data.frame.select(
        nw.col(data.key).is_between(min_value, max_value, closed=closed)
    )
```

**`notin` — use tilde, NOT `.not_()`:**
```python
def notin(data, forbidden_values):
    return data.frame.select(~nw.col(data.key).is_in(forbidden_values))
```

**`str_contains` — remove `literal=False`:**
```python
def str_contains(data, pattern):
    pattern = pattern.pattern if isinstance(pattern, re.Pattern) else pattern
    return data.frame.select(nw.col(data.key).str.contains(pattern))
```

**`str_matches` — anchor with ^ (same as Polars, but no `literal=` param):**
```python
def str_matches(data, pattern):
    pattern = pattern.pattern if isinstance(pattern, re.Pattern) else pattern
    if not pattern.startswith("^"):
        pattern = f"^{pattern}"
    return data.frame.select(nw.col(data.key).str.contains(pattern))
```

**`str_length` — use nw.col().str.len_chars(), same 4 cases as Polars:**
```python
def str_length(data, min_value=None, max_value=None, exact_value=None):
    n_chars = nw.col(data.key).str.len_chars()
    if exact_value is not None:
        return data.frame.select(n_chars.eq(exact_value))
    if min_value is None and max_value is None:
        raise ValueError("Must provide at least one of 'min_value' and 'max_value'")
    elif min_value is None:
        return data.frame.select(n_chars.le(max_value))
    elif max_value is None:
        return data.frame.select(n_chars.ge(min_value))
    else:
        return data.frame.select(n_chars.is_between(min_value, max_value, closed="both"))
```

**DO NOT include `unique_values_eq`** — that is a Polars-specific builtin that forces materialization; it is not part of the 14 narwhals builtins.
