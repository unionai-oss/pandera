```python
def _materialize(frame) -> nw.DataFrame:
    if isinstance(frame, nw.LazyFrame):
        return frame.collect()
    native = nw.to_native(frame)
    if hasattr(native, "execute"):
        return nw.from_native(native.execute())
    return frame
```
